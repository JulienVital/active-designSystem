# Render

```ts
//template
<template>
    <ul>
    <li v-for="currentLi in props.items" @contextmenu="onRightClick($event, currentLi.id)">
  </ul>
    <ContextMenu ref="menu" :model="items" @hideMenu="hide" />
</template>

```
```ts
//script
<script setup>
import { ref } from 'vue';

const menu = ref(null)
const selectedLi = ref(null)

const onRightClick = (event, id) => {
    selectedLi.value = id;
    menu.value.show(event);
};

const hide = ()=>{
  selectedLi.value = null;
}
const item = [
  {
    label: 'Option 1',
    command: () => {
      console.log('Option 1 ' + selectedLi.value)
    }
  },
  {
    label: 'Option 2',
    command: () => {
      console.log('Option 2' + selectedLi.value)
    }
  }
]
</script>
```
